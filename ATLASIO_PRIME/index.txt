function toggleRec(){
    const btn = document.getElementById('btn-rec');

        isRecording = true;

}

['sin','cos','tan','log','π','7','8','9','/','√','4','5','6','*','xʸ','1','2','3','-','(','C','0','.','+',')'].forEach(k=>{
    const b=document.createElement('div');b.className='c-btn';b.innerText=k;
    if(['sin','cos','tan','log','√','π','xʸ'].includes(k)) b.style.color='#d4af37';
    if(k==='C') b.style.color='#ff3333';
    b.onclick=()=>{
        if(k==='C') ex='';
        else if(k==='π') ex+='Math.PI';
        else if(k==='sin'||k==='cos'||k==='tan') ex+=`Math.${k}(`;
        else if(k==='log') ex+='Math.log10(';
        else if(k==='√') ex+='Math.sqrt(';
        else if(k==='xʸ') ex+='**';
        else ex+=k;
        document.getElementById('scr').innerText=ex||'0';
    };
    document.getElementById('keys').appendChild(b);
});
